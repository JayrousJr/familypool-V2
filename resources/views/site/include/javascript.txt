
const validation = new JustValidate("#app-form");

validation
.addField("#fname", [
	{
		rule: "required",
		errorMessage: "First Name is required",
	}
])
.addField("#sname", [
	{
		rule: "required",
		errorMessage: "Second Name is Required",
	}
])
.addField("#streetAddress",[
	{
		rule: "required",
		errorMessage: "Street Address is required"
	}
])
.addField("#city",[
	{
		rule: "required",
		errorMessage: "City is required",
	}
])
.addField("#state",[
	{
		rule: "required",
		errorMessage: "State / Province is required",
	}
])
.addField("#zip",[
	{
		rule: 'required',
		errorMessage: "Postal / Zip Code is Required",
	}
])

.addField("#email",[
	{
		rule: 'required',
		errorMessage: "Email is required",
	},
	{
		rule: "email",
		errorMessage: "Invalid e-mail format"
	}
])
.addField("#phone",[
	{
		rule: "required",
		errorMessage: "Phone number is required",
	},{
		rule: "minLength",
		value: 10,
	},{
		rule: "maxLength",
		value: 12,
	}
])
.addField("#birthdate",[
	{
		rule: "required",
		errorMessage: "Birth date is required ",
	}
])

.addRequiredGroup("#ageCheckboxes", 'Select YES or NO')
.addRequiredGroup("#licenceCheckboxes", 'Select YES or NO')
.addRequiredGroup("#transportOwnership", 'Select YES or NO')

.addField("#work-hour",[
	{
		rule: "required",
		errorMessage: "Working hours field is required"
	},
	{
		rule: "minNumber",
		value:  20,
		errorMessage: "Working hours per week should be greater than 20"
	},{
		rule: "maxNumber",
		value: 168,
		errorMessage: "Working hour per week should be less than 168"
	}
])
.addRequiredGroup("#daysAvailable", 'Choose the you are available')

.addField('#cvFile', [
   {
    rule: 'minFilesCount',
    value: 1,
	errorMessage: "Attach your CV in a .pdf file format",
  },
	{
    rule: 'files',
    value: {
      files: {
        extensions: ['pdf'],
        maxSize: 1000000,
        type: ['document/pdf'],
        name: ['cv.pdf'],
      },
    },
		errorMessage: "File attached is either not a .pdf file or is larger than 1Mb"
  },
 
])
.addField('#photo', [
   {
    rule: 'minFilesCount',
    value: 1,
	errorMessage: "Attach passport in a .jpg, .png or .jpeg file format",
  },
	{
    rule: 'files',
    value: {
      files: {
         extensions: ['jpg', 'png'],
        maxSize: 1000000,
        type: ['image/jpg', 'image/png'],
        name: ['file1.jpg', 'file2.png'],
      },
    },
		errorMessage: "File entered is either not a .jpg , .png or .jpeg file or is larger than 1Mb"
  },
 
])
.onSuccess((event) => {
	document.getElementById("app-form").submit();
});
